118 {
119  register bool old;
120  __asm__ __volatile__("MRS %0, PRIMASK" : "=r" (old));
121  __asm__ __volatile__("" : : : "memory");
122  __asm__ __volatile__("MSR PRIMASK, %0" : : "r" (mask));
