#define RTC_CR_ADD1H
值:
(1<<16) /* RTC_CR_ADD1H: Add 1 hour (summer
time change) */
#define RTC_CR_ALRAIE
值:
(1<<12) /* RTC_CR_ALRAIE: Alarm A interrupt
enable */
#define RTC_CR_ALRBIE
值:
(1<<13) /* RTC_CR_ALRBIE: Alarm B interrupt
enable */
#define RTC_CR_BYPSHAD
值:
(1<<5) /* RTC_CR_BYPSHAD: Bypass the shadow
registers */
#define RTC_CR_COSEL
值:
(1<<19) /* RTC_CR_COSEL: Calibration output
selection */
#define RTC_CR_DCE
值:
(1<<7) /* RTC_CR_DCE: Course digital
calibration enable */
#define RTC_CR_REFCKON
值:
(1<<4) /* RTC_CR_REFCKON: Reference clock
detection enable */
#define RTC_CR_SUB1H
值:
(1<<17) /* RTC_CR_SUB1H: Subtract 1 hour
(winter time change) */
#define RTC_CR_TSEDGE
值:
(1<<3) /* RTC_CR_TSEDGE: Timestamp event
active edge */
#define RTC_CR_TSIE
值:
(1<<15) /* RTC_CR_TSIE: Timestamp interrupt
enable */
#define RTC_CR_WUTIE
值:
(1<<14) /* RTC_CR_WUTIE: Wakeup timer
interrupt enable */
#define RTC_ISR_TAMP3F
值:
(1<<15) /* TAMP3F: TAMPER3 detection flag
(not on F4)*/
