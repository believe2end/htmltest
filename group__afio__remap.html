宏定义说明

#define AFIO_MAPR_ADC1_ETRGINJ_REMAP   (1 << 17)

ADC1 external trigger injected conversion remapping (only low-, medium-, high- and XL-density devices)

#define AFIO_MAPR_ADC1_ETRGREG_REMAP   (1 << 18)

ADC1 external trigger regulator conversion remapping (only low-, medium-, high- and XL-density devices)

#define AFIO_MAPR_ADC2_ETRGINJ_REMAP   (1 << 19)

ADC2 external trigger injected conversion remapping (only low-, medium-, high- and XL-density devices)

#define AFIO_MAPR_ADC2_ETRGREG_REMAP   (1 << 20)

ADC2 external trigger regulator conversion remapping (only low-, medium-, high- and XL-density devices)

#define AFIO_MAPR_I2C1_REMAP   (1 << 1)

I2C1 remapping

#define AFIO_MAPR_PD01_REMAP   (1 << 15)

Port D0/Port D1 mapping on OSC_IN/OSC_OUT

#define AFIO_MAPR_SPI1_REMAP   (1 << 0)

SPI1 remapping

#define AFIO_MAPR_TIM4_REMAP   (1 << 12)

TIM4 remapping

#define AFIO_MAPR_TIM5CH4_IREMAP   (1 << 16)

TIM5 channel 4 internal remap

#define AFIO_MAPR_USART1_REMAP   (1 << 2)

USART1 remapping

#define AFIO_MAPR_USART2_REMAP   (1 << 3)

USART2 remapping
